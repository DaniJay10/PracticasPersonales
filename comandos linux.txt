cd ~/pb-backend && cat > Dockerfile <<'EOF'
FROM debian:stable-slim
WORKDIR /app
RUN apt-get update && apt-get install -y --no-install-recommends ca-certificates bash && rm -rf /var/lib/apt/lists/*
COPY pocketbase /app/pocketbase
COPY render-start.sh /app/render-start.sh
RUN chmod +x /app/pocketbase /app/render-start.sh && mkdir -p /app/pb_data /app/pb_logs
CMD ["bash","-lc","./render-start.sh"]
EOF
git add Dockerfile && git commit -m "chore: add Dockerfile for Render" && git push origin main
